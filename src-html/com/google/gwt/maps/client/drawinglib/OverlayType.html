001    package com.google.gwt.maps.client.drawinglib;
002    
003    /**
004     * The types of overlay that may be created by the DrawingManager.
005     * <br><br>
006     * See <a href="https://developers.google.com/maps/documentation/javascript/reference#OverlayType">OverlayType API Doc</a>
007     */
008    public enum OverlayType {
009    
010      /**
011       * Specifies that the DrawingManager creates circles, and that the overlay given in the overlaycomplete event is a circle.
012       */
013      CIRCLE,
014      
015      /**
016       * Specifies that the DrawingManager creates markers, and that the overlay given in the overlaycomplete event is a marker.
017       */
018      MARKER,
019      
020      /**
021       * Specifies that the DrawingManager creates polygons, and that the overlay given in the overlaycomplete event is a polygon.
022       */
023      POLYGON,
024      
025      /**
026       * Specifies that the DrawingManager creates polylines, and that the overlay given in the overlaycomplete event is a polyline.
027       */
028      POLYLINE,
029      
030      /**
031       * Specifies that the DrawingManager creates rectangles, and that the overlay given in the overlaycomplete event is a rectangle.
032       */
033      RECTANGLE;
034      
035      
036      public String value() {
037        return name().toLowerCase();
038      }
039      
040      public static OverlayType fromValue(String type) {
041        return valueOf(type.toUpperCase());
042      }
043    
044      public String toString() {
045        return name();
046      }
047      
048    }
