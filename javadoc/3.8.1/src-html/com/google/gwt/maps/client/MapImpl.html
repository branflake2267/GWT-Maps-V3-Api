056/**
057 * Creates a new map inside of the given HTML container, which is typically a
058 * DIV element <br>
059 * <br>
060 * See <a href=
061 * "https://developers.google.com/maps/documentation/javascript/reference#Map"
062 * >Map API Doc</a>
063 */

067         * Creates a new map inside of the given HTML container, which is typically
068         * a DIV element.
069         */
070        protected MapImpl() {
071        }

073        /**
074         * Creates a new map inside of the given HTML container, which is typically
075         * a DIV element.
076         * 
077         * @param element
078         *            - map dom element container, like a div
079         * @param options
080         *            - {@link MapOptions}
081         * @return {@link MapImpl}
082         */
083        public final static MapImpl newInstance(Element element, MapOptions options) {
084                return createJso(element, options);
085        }

095                        MapOptions options) /*-{
096                return new $wnd.google.maps.Map(element, options);

222        /**
223         * Pans the map by the minimum amount necessary to contain the given
224         * LatLngBounds. It makes no guarantee where on the map the bounds will be,
225         * except that as much of the bounds as possible will be visible. The bounds
226         * will be positioned inside the area bounded by the map type and navigation
227         * (pan, zoom, and Street View) controls, if they are present on the map. If
228         * the bounds is larger than the map, the map will be shifted to include the
229         * northwest corner of the bounds. If the change in the map's position is
230         * less than both the width and height of the map, the transition will be
231         * smoothly animated.
232         * 
233         * @param latLngBounds
234         */
